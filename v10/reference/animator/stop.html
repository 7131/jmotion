Animator.prototype.stop()

The stop() method stops the animation. Nothing happens unless the animation is running.

Syntax

stop()

Parameters

none

Return value

none

Examples

const core = new jmotion.Core("#board");
const animator = new jmotion.Animator(core);

const pl0 = { "x":   0, "y": -30 };
const pl1 = { "x": -30, "y":   0 };
const pl2 = { "x":   0, "y":  30 };
const pl3 = { "x":  30, "y":   0 };

animator.props = [
    {
        "init": [],
        "loop": [ pl0, pl1, pl2, pl3 ],
    }
];
animator.start(0.1);
setTimeout(animator.stop.bind(animator), 2000);
